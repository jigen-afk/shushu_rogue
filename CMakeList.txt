cmake_minimum_required(VERSION 3.10)
project(TextSpire)

set(CMAKE_CXX_STANDARD 17)

# 包含原本的 Classes 目录，但要剔除不需要的文件
include_directories(src)

# 手动指定需要的源文件，不要用 file(GLOB) 以免把垃圾文件包进去
set(SOURCES
    src/main.cpp
    src/Card.cpp
    src/Player.cpp
    src/Monster.cpp
    src/CombatSystem.cpp
    src/RandomGenerator.cpp
    # ... 添加其他筛选后的文件
)

add_executable(TextSpire ${SOURCES})
